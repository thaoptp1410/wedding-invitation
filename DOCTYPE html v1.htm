    // Smooth scroll

    document.querySelectorAll('a[href^="#"]').forEach(a=>{

      a.addEventListener('click', e=>{

        const id = a.getAttribute('href');

        const el = document.querySelector(id);

    // Countdown (đặt ngày cưới tại đây: YYYY-MM-DDTHH:mm:ss)

    const eventDate = new Date('2025-11-23T11:30:00');

    const cdWrap = document.getElementById('countdown');

    function updateCountdown(){

      const now = new Date();

      let diff = Math.max(0, eventDate - now);

      const d = Math.floor(diff/86400000); diff -= d*86400000;

      const h = Math.floor(diff/3600000); diff -= h*3600000;

      const m = Math.floor(diff/60000); diff -= m*60000;

      const s = Math.floor(diff/1000);

      cdWrap.innerHTML = [

        ['Ngày', d], ['Giờ', h], ['Phút', m], ['Giây', s]

      ].map(([label,val])=>`<div class="cd-item"><b>${String(val).padStart(2,'0')}</b><span>${label}</span></div>`).join('');

    }

    updateCountdown(); setInterval(updateCountdown, 1000);

 

    // Petals animation

    const PETALS = 16;

    function spawnPetal(){

      const p = document.createElement('div');

      p.className = 'petal';

      const startX = Math.random()*100; // vw

      const dur = 8000 + Math.random()*6000;

      p.style.left = startX+'vw';

      p.style.animationDuration = dur+'ms';

      p.style.transform = `translateY(0) rotate(${Math.random()*360}deg)`;

      document.body.appendChild(p);

      setTimeout(()=>p.remove(), dur);

    }

    setInterval(spawnPetal, 700);

 

    // Music toggle

    const btn = document.getElementById('musicBtn');

    const bgm = document.getElementById('bgm');

    btn.addEventListener('click', async ()=>{

      try{

        if(bgm.paused){ await bgm.play(); btn.setAttribute('aria-pressed','true'); btn.textContent='❚❚'; }

        else { bgm.pause(); btn.setAttribute('aria-pressed','false'); btn.textContent='♪'; }

      }catch(err){ alert('Không phát được nhạc: ' + err.message); }

    });

 

    // RSVP handler (demo – gửi qua mailto)

    function rsvpSubmit(e){

      e.preventDefault();

      const name = document.getElementById('name').value.trim();

      const phone = document.getElementById('phone').value.trim();

      const guests = document.getElementById('guests').value;

      const msg = document.getElementById('message').value.trim();

      const body = encodeURIComponent(`Họ tên: ${name}\nĐiện thoại: ${phone}\nKhách: ${guests}\nLời nhắn: ${msg}`);

      window.location.href = `mailto:lanminh.wed@example.com?subject=RSVP - ${encodeURIComponent(name)}&body=${body}`;

      document.getElementById('rsvpResult').textContent = 'Cảm ơn bạn! Hệ thống đã mở email để gửi xác nhận.';

    }
